


`timescale 1ns / 1ps

module soru1a( 
    // Inputs and outputs is defined here.
    
    input A,B,C,
    output F,Q 

    );
    
    wire nA;
    not i1(nA,A);
    
    wire nAxorB;
    xor i2(nAxorB,nA,B);
    
    wire BnandC;
    nand i3(BnandC,B,C);
    
    wire nC;
    not i4 (nC,C);
    
    wire nCorB;
    or i5(nCorB,nC,B);
    
    xnor(F,nAxorB,BnandC);
    
    nor(Q,nAxorB,BnandC,nCorB);
    
    
    
    
endmodule
